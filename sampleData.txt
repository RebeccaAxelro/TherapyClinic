
#Create Tables
#	Client
CREATE TABLE Client (
	ClientID int,
	Name varchar(255),
	Address varchar(255),
	Email varchar(255),
	PhoneNumber int
	PRIMARY KEY(ClientID)
);
# Therapist
CREATE TABLE Therapist (
	TherapistID int,
	Name varchar(255),
	Email varchar(255),
	PRIMARY KEY(TherapistID)
);
# Diagnosed
CREATE TABLE Diagnosed (
	ClientID int,
	DisorderCode int,
	PRIMARY KEY(ClientID),
	PRIMARY KEY(DisorderCode)
);
# Specializes
CREATE TABLE Specializes (
	TherapistID int,
	DisorderCoder int,
	PRIMARY KEY(TherapistID),
	PRIMARY KEY(DisorderCode)
);
# Disorder
CREATE TABLE Disorder (
	DisorderCode int,
	DisorderName varchar(255),
	PRIMARY KEY(DisorderCode)
);
# Treated
CREATE TABLE Treated (
	ClientID int,
	AppointmentID int,
	PRIMARY KEY(ClientID),
	PRIMARY KEY(AppointmentID)
);
# Treating
CREATE TABLE Treating (
	TherapistID int,
	AppointmentID int,
	PRIMARY KEY(TherapistID),
	PRIMARY KEY(AppointmentID)
);
# Appointment
CREATE TABLE Appointment (
	AppointmentID int,
	TreatmentID int 
	Date varchar(255),
	Room_no varchar(255),
	Duration int,
	PRIMARY KEY(AppointmentID),
	PRIMARY KEY(TreatmentID)  
);
# Treatment
CREATE TABLE Treatment (
	TreatmentID int,
	TreatmentName varchar(255),
	PRIMARY KEY(TreatmentID) 
);
